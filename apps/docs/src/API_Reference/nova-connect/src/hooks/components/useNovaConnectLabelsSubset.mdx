## Example

```typescript
function ConnectModal() {
  const { connectWallet, connecting, disconnect } = useNovaConnectLabelsSubset(['connectWallet', 'connecting', 'disconnect']);

  return (
    <div>
      <h2>{connectWallet}</h2>
      <span>{connecting}</span>
      <button>{disconnect}</button>
    </div>
  );
}
```
